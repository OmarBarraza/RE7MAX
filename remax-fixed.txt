<script>

const API_BASE='http://localhost:3000/api';

window.addEventListener("scroll",function(){
const navbar=document.querySelector(".navbar");
if(window.scrollY>80){
navbar.classList.add("scrolled");
}else{
navbar.classList.remove("scrolled");
}
});

async function fetchAllProperties(){

document.getElementById('loading').style.display='block';
document.getElementById('results').innerHTML='';

try{

const response=await fetch(`${API_BASE}/propiedades`);
const data=await response.json();

let properties=[];

if(Array.isArray(data)) properties=data;
else if(data.data) properties=data.data;
else if(data.propiedades) properties=data.propiedades;

renderProperties(properties);

}catch(error){

renderProperties(MOCK_PROPERTIES);

}

document.getElementById('loading').style.display='none';

}

function renderProperties(properties){

const results=document.getElementById('results');

results.innerHTML=`

<div class="property-grid">

${properties.map(prop=>{

const image=prop.imagen || prop.image || prop.foto || 'https://picsum.photos/400/250';

const price=formatPrice(prop.precio || prop.price || 0);

const title=prop.titulo || prop.title || 'Propiedad';

const location=prop.ubicacion || prop.location || 'Ubicación no disponible';

const barWidth=Math.floor(Math.random()*60+20);

return`

<div class="property-card">

<div class="property-image-wrap">

<img src="${image}" class="property-image">

<div class="remax-badge">RE/MAX</div>

</div>

<div class="property-body">

<div class="property-title">${title}</div>

<div class="property-address">📍 ${location}</div>

<div class="property-price">${price}</div>

<div class="price-bar-wrap">
<div class="price-bar" style="width:${barWidth}%"></div>
</div>

</div>

</div>

`;

}).join('')}

</div>

`;

}

function formatPrice(price){

return new Intl.NumberFormat('es-MX',{
style:'currency',
currency:'MXN',
minimumFractionDigits:0
}).format(price);

}

document.addEventListener('DOMContentLoaded',fetchAllProperties);

</script>
