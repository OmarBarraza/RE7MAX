async function fetchAllProperties(){
    document.getElementById('loading').style.display = 'block';
    document.getElementById('results').innerHTML = '';

    try {
        const response = await fetch(`${API_BASE}/propiedades`);
        const data = await response.json();

        let properties = [];
        if(Array.isArray(data)) properties = data;
        else if(data.data) properties = data.data;
        else if(data.Data) properties = data.Data;
        else if(data.propiedades) properties = data.propiedades;

        renderProperties(properties);
    } catch(error) {
        // Use mock data if API is unavailable
        renderProperties(MOCK_PROPERTIES);
    }

    document.getElementById('loading').style.display = 'none';
}

function renderProperties(properties){
    const results = document.getElementById('results');

    if(!properties || properties.length === 0){
        results.innerHTML = '<p style="color:var(--text-light);text-align:center;padding:2rem;">No se encontraron propiedades</p>';
        return;
    }

    results.innerHTML = `
    <div class="property-grid">
        ${properties.map(prop => {
            const image = prop.imagen || prop.image || prop.foto || `https://picsum.photos/seed/${prop.id || Math.random()}/600/400`;
            const price = formatPrice(prop.precio || prop.price || 0);
            const title = prop.titulo || prop.title || 'Propiedad';
            const agente = prop.agente || '';
            const location = prop.ubicacion || prop.location || 'Ubicación no disponible';
            const recamaras = prop.recamaras;
            const banos = prop.banos;
            const m2 = prop.m2;
            const agenteName = prop.agenteName || 'Agente RE/MAX';
            const avatarInicial = prop.avatarInicial || agenteName.charAt(0);
            const barWidth = prop.barWidth || Math.floor(Math.random() * 60 + 20);

            const featuresHTML = (recamaras || banos || m2) ? `
            <div class="property-features">
                ${recamaras ? `<span class="feature">🛏 ${recamaras}</span>` : ''}
                ${banos ? `<span class="feature">🚿 ${banos}</span>` : ''}
                ${m2 ? `<span class="feature">📐 ${m2} m²</span>` : ''}
            </div>` : '';

            return `
            <div class="property-card">
                <div class="property-image-wrap">
                    <img src="${image}" class="property-image" alt="${title}" loading="lazy">
                    <div class="remax-badge">RE/MAX</div>
                </div>
                <div class="property-body">
                    <div class="property-title">${title}</div>
                    ${agente ? `<div class="property-agent"><div class="agent-dot"></div>${agente}</div>` : ''}
                    <div class="property-address">📍 ${location}</div>
                    <div class="property-price">${price}</div>
                    <div class="price-bar-wrap">
                        <div class="price-bar" style="width:${barWidth}%"></div>
                    </div>
                </div>
                ${featuresHTML}
                <div class="property-footer">
                    <div class="agent-info">
                        <div class="agent-avatar">${avatarInicial}</div>
                        <span class="agent-name">${agenteName}</span>
                    </div>
                    <div class="card-actions">
                        <button class="action-btn" title="Compartir">↗</button>

                </div>

        }).join('')}
    </div>`;
}

function formatPrice(price){
    if(!price || price === 0) return 'Precio no disponible';
    return new Intl.NumberFormat('es-MX', {
        style:'currency',
        currency:'MXN',
        minimumFractionDigits:0
    }).format(price);
}
